📋 Python Cheatsheet

String Methods

s = "hello world"
s.upper()           # "HELLO WORLD"
s.lower()           # "hello world"
s.capitalize()      # "Hello world"
s.split()           # ['hello', 'world']
s.replace("o", "0") # "hell0 w0rld"
s.strip()           # Remove leading/trailing whitespace
s.startswith("he")  # True
s.endswith("ld")    # True
s.find("wor")       # 6 (index)
s.count("l")        # 3 (occurrences)

List Operations

lst = [1, 2, 3, 4, 5]
lst.append(6)           # Add to end
lst.extend([7, 8])      # Add multiple items
lst.insert(0, 0)        # Insert at index
lst.remove(3)           # Remove by value
lst.pop()               # Remove and return last item
lst.pop(0)              # Remove and return at index
lst.sort()              # Sort in-place
lst.reverse()           # Reverse in-place
lst.clear()             # Remove all items
len(lst)                # Get length
lst.index(2)            # Find index of value
lst.count(3)            # Count occurrences

Dictionary Operations

d = {"name": "John", "age": 30}
d["name"]           # Access value
d.get("name")       # Safe access (returns None if key doesn't exist)
d.get("city", "NY") # Default value
d["city"] = "NYC"   # Add/update key
del d["age"]        # Delete key
d.keys()            # Get all keys
d.values()          # Get all values
d.items()           # Get key-value pairs
d.pop("name")       # Remove and return value
d.clear()           # Remove all items

List Comprehension

# Basic comprehension
[x*2 for x in range(5)]           # [0, 2, 4, 6, 8]

# With condition
[x for x in range(10) if x % 2 == 0]  # [0, 2, 4, 6, 8]

# Nested comprehension
[[x*y for x in range(3)] for y in range(3)]

# Dictionary comprehension
{x: x**2 for x in range(5)}       # {0: 0, 1: 1, 2: 4, 3: 9, 4: 16}

# Set comprehension
{x for x in range(5)}             # {0, 1, 2, 3, 4}

Functions & Lambda

# Function definition
def greet(name):
    return f"Hello, {name}!"

# Lambda function
square = lambda x: x ** 2
square(5)  # 25

# Map function
nums = [1, 2, 3, 4, 5]
squared = list(map(lambda x: x**2, nums))  # [1, 4, 9, 16, 25]

# Filter function
evens = list(filter(lambda x: x % 2 == 0, nums))  # [2, 4]

Exception Handling

try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero")
except Exception as e:
    print(f"Error: {e}")
finally:
    print("Cleanup code here")

File Operations

# Read file
with open("file.txt", "r") as f:
    content = f.read()      # Read all
    lines = f.readlines()   # Read lines

# Write file
with open("file.txt", "w") as f:
    f.write("Hello")

# Append to file
with open("file.txt", "a") as f:
    f.write("\nNew line")

# Check if file exists
import os
os.path.exists("file.txt")
